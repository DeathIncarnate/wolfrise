class NaziBullet : FastProjectile
{
	Default
	{
		RenderStyle "Add";
		Obituary "$OB_MPROCKET";
		Radius 8;
		Height 8;
		Speed 72;	// 32
		Damage 4;
		Scale 0.32;
		MissileType "BulletTrail";
		//SeeSound "DSMG42";
		Projectile;
	}
	
	States
	{
		Spawn:
			BULL A 1 BRIGHT;
			Loop;
		Death:
			PUFF BCD 5 Bright A_SetScale(0.8, 0.8);
			Stop;
	}
}
