class KnifePuff1 : BulletPuff
{
	Default
	{
		AttackSound "weapons/knife/impact";
		Gravity 3.2;
		Speed 4;
		VSpeed 4;
		Mass 32;
		Projectile;
		+NOBLOCKMAP
		+NOEXTREMEDEATH 
		+FORCEXYBILLBOARD
		+DONTSPLASH
		-NOGRAVITY
	}
	
	States
	{
		Spawn:
			TNT1 A 0 A_PlaySound("weapons/knife/impact");
			PUFF ABCD 1 BRIGHT;
			Stop;

			PUFF ABCD 1 BRIGHT;
			Stop;
		Death:
			PUFF ABCD 1 BRIGHT;
			Stop;
		XDeath:
			PUFF ABCD 1 BRIGHT;
			Stop;
	}
}

class BloodPuff1 : Actor	// MEDIUM BLOOD
{
	Default
	{
		Health 270;
		RenderStyle "Normal";
		Alpha 1.0;
		Scale 0.72;	// 0.56
		Gravity 0.88;
		Speed 1;
		VSpeed 1;
		ReactionTime 250;
		+RANDOMIZE
		+Missile
		+NoBlockmap
		-SOLID
	}
	
	States
	{
		Spawn:
			TNT1 A 0 A_Jump(255, "Spawn2");
		Spawn2:
			BPUF AAAAAAAA 4 A_CountDown;
			Loop;
		Death:
			BPUF AAAAAA 2 BRIGHT A_FadeOut(0.024);
			Stop;
	}
}
