class PatrolGuard : Actor
{
	Default
	{
		//$Category Enemies
		obituary "%o was shot by a Patrol Guard.";
		Health 20;
		Radius 21;
		Height 72;
		Mass 300;
		Speed 12;
		dropitem "SmallAmmo";
		Painchance 100;
		Maxtargetrange 1024;
		MinMissileChance 2;
		DeathSound "enemies/death2";
		SeeSound "enemies/swein";
		PainSound "UBERPAIN";
		BloodType "BloodPuff1";
		Scale 1.44;
		Monster;
		+MISSILEMORE
		+MISSILEEVENMORE
	}
	
	States
	{
		Spawn:
			WHIT A 2 A_Look;
			Loop;
		See:
			WHIT BCDE 4 A_Chase;
			Loop;
		Missile:
			WHIT F 16 A_FaceTarget;
			WHIT G 4;
			TNT1 A 0 A_PlaySound("weapons/luger/fire");
			WHIT H 4 A_CustomBulletAttack(8,0,1,4);
			Goto See;
		Pain:
			WHIT U 2 A_Pain;
			Goto See;
		Death:
			TNT1 A 0 A_GiveInventory("WolfScore", 1000, AAPTR_PLAYER1);
			WHIT U 4;
			WHIT V 4 A_Scream;
			WHIT W 4 A_NoBlocking;
			WHIT WXY 4;
			WHIT Z -1;
			Stop;
	}
}
