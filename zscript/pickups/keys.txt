class WolfKey : Key
{
	Default
	{
		//$Category Wolfenstein / Keys
		Inventory.PickUpSound "PICKKEY";
		Scale 1.6;
	}
}

class WolfGoldKey : WolfKey
{
	Default
	{
		Inventory.PickupMessage "You got the Gold Key!";
		Inventory.Icon "GKEYICON";
	}
	
	States
	{
		Spawn:
			GKEY A 5;
			GKEY A 5 Bright;
			Loop;
	}
}
