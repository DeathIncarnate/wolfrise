class KnifePuff1 : BulletPuff
{
	Default
	{
		AttackSound "weapons/knife/impact";
		Gravity 3.2;
		Speed 4;
		VSpeed 4;
		Mass 32;
		Projectile;
		+NOBLOCKMAP
		+NOEXTREMEDEATH 
		+FORCEXYBILLBOARD
		+DONTSPLASH
		-NOGRAVITY
	}
	
	States
	{
		Spawn:
			TNT1 A 0 A_PlaySound("weapons/knife/impact");
			PUFF ABCD 1 BRIGHT;
			Stop;
		Melee:
			PUFF ABCD 1 BRIGHT;
			Stop;
		Death:
			PUFF ABCD 1 BRIGHT;
			Stop;
		XDeath:
			PUFF ABCD 1 BRIGHT;
			Stop;
	}
}
