	action void A_SpearStrike()
	{
		A_DamageSelf(6, "SpearDrain");
		A_Saw("eapons/spear/attack", "eapons/spear/attack", 666, "Explosion");
		A_PlaySound("weapons/spear/attack");
	}

		DamageType "SpearDrain";

		Weapon.BobRangeY 0.5;

		Tag "The Spear of Destiny";
		+WEAPON.NOAUTOFIRE;
		+WEAPON.NOALERT;
		+WEAPON.NOAUTOAIM;
	}

				A_Recoil(-16.0);
