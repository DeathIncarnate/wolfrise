class Officer : Actor
{
	Default
	{
		//$Category Enemies
		obituary "%o was hunted by an officer";
		Health 75;
		Radius 21;
		Height 72;
		Mass 300;
		Speed 16;
		dropitem "SmallAmmo";
		Painchance 96;
		Maxtargetrange 2048;
		MinMissileChance 2;
		DeathSound "enemies/death2";
		SeeSound "enemies/halt";
		PainSound "UBERPAIN";
		BloodType "BloodPuff1";
		Scale 1.44;
		Monster;
		+MISSILEMORE
		+MISSILEEVENMORE
	}
	
	States
	{
		Spawn:
			OFF1 A 2 A_Look;
			Loop;
		See:
			OFF1 BCDE 3 A_Chase;
			Loop;
		Missile:
			OFF1 F 2 A_FaceTarget;
			OFF1 G 12;
		Firing:
			OFF1 H 4
			{
				A_PlaySound("weapons/luger/softfire");
				A_CustomBulletAttack(0,0,1,2);
			}
			OFF1 G 8;
			TNT1 A 0 A_CPosRefire;
			Goto Firing;
		Pain:
			OFF1 O 2 A_Pain;
			Goto See;
		Death:
			TNT1 A 0 A_GiveToTarget("WolfScore", 1000);
			OFF1 I 4 A_Scream;
			OFF1 J 4 A_NoBlocking;
			OFF1 KLM 4;
			OFF1 N -1;
			Stop;
	}
}
