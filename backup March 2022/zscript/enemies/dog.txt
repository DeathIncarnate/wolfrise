class Dog : Actor
{
	Default
	{
		//$Category Enemies
		ReactionTime 4;
		obituary "%o was devoured by a dog.";
		radius 25;
		height 56;
		Health 14;
		mass 16;
		painchance 0;
		Speed 28;
		Scale 1.36;
		MeleeRange 96;
		seesound "DOG1BARK";
		deathsound "DOG1DETH";
		MONSTER;
		+MISSILEMORE
		+MISSILEEVENMORE
		+NOINFIGHTING
		+DONTHARMSPECIES
		-CANUSEWALLS
	}
	
	States
	{
		Spawn:
			DOG1 A 2 A_Look;
			Loop;
		See:
			DOG1 AABBCCDD 3 A_Chase;
			Loop;
		Melee:
			DOG1 I 2 A_FaceTarget;
			DOG1 J 3;
			TNT1 A 0 A_CustomMeleeAttack(10, "DOG1BITE");
			DOG1 K 3;
			Goto See;
		Death:
			TNT1 A 0 A_GiveToTarget("WolfScore", 200);
			DOG1 E 7;
			DOG1 F 7 A_Scream;
			DOG1 G 7 A_NoBlocking;
			DOG1 H -1;
			Stop;
	}
}
