class BJSphere : Health
{
	Default
	{
		//$Category Powerups
		Height 12;
		Radius 10;
		Inventory.Amount 100;
		Inventory.MaxAmount 200;
		Inventory.PickupMessage "You absorbed a power sphere (+100 health)";
		Inventory.pickupsound "POWRPICK";
		Scale 1.2;
		+FLOATBOB
		+NOGRAVITY
	}
	
	States
	{
		Spawn:
			PWR1 ABCDEFGFEDCB 2 Bright;
			Loop;
	}
}
